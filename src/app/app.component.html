Install

1. Install from NPM

npm install ngx-modal --save

2.Add to import Modules and Add to entry entryComponents in @NgModules, the modules that you want to put inside the modal.

          
@NgModule({{ '{' }}
  declarations: [
    AppComponent,
    SampleModalComponent,
  ],
  entryComponents: [
    SampleModalComponent
  ],
  imports: [
    BrowserModule,
    FormsModule,
    NgxModalModule,
  ],
  providers: [],
  bootstrap: [AppComponent]
})

3. Import your Service in a component and Create a new Modal.

          
export class AppComponent {{ '{' }}

  constructor(private modalService: NgxModalService) {{ '{' }}

  }

  openModal(){{ '{' }}
    const params = {{ '{' }}};
    this.modalService.create(SampleModalComponent, params).then((modal: NgxModal) => {{ '{' }}
      modal.open();
    });
  }

}

in params, you can add all params that you have into your component for init with values.
